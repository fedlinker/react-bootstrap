## Button

The button component is used for the area of clicking.

## Demos

### Types

<Playground>
  {() => {
    var margin = "12px";
    return [
      "primary",
      "success",
      "warning",
      "danger",
      "info",
      "light",
      "dark",
      "link",
    ].map(t => {
      return (

        </Button>
      );

  }}
</Playground>

### Loading state

<Playground>
  {() => {
    const [loading, setLoading] = useState(true);
    var margin = "12px";
    const handleLoadingChange = () => {
      setLoading(!loading);
    };
    return (
      <div>
        <div>
          Button status: {loading ? "Loading" : "Not loading"}{" "}
          <Button onClick={handleLoadingChange}>Change Status</Button>
        </div>
        <div>
          {[
            "primary",
            "success",
            "warning",
            "danger",
            "info",
            "light",
            "dark",
            "link",
          ].map(t => {
            return (
              <Button type={t} key={t} style={{ margin }} loading={loading}>
                {t.toUpperCase()}
              </Button>
            );
          })}
        </div>
      </div>
    );
  }}
</Playground>
