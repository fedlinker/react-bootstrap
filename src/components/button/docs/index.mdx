### Loading state

<Playground>
  {() => {
    const [loading, setLoading] = useState(true);
    var margin = "12px";
    const handleLoadingChange = () => {
      setLoading(!loading);
    };
    return (
      <div>
        <div>
          Button status: {loading ? "Loading" : "Not loading"}{" "}
          <Button onClick={handleLoadingChange}>Change Status</Button>
        </div>
        <div>
          {[
            "primary",
            "success",
            "warning",
            "danger",
            "info",
            "light",
            "dark",
            "link",
          ].map(t => {
            return (
              <Button type={t} key={t} style={{ margin }} loading={loading}>
                {t.toUpperCase()}
              </Button>
            );
          })}
        </div>
        <div>
          {[
            "primary",
            "success",
            "warning",
            "danger",
            "info",
            "light",
            "dark",
            "link",
          ].map(t => {
            return (
              <Button
                type={t}
                key={t}
                style={{ margin }}
                loading={loading}
                outline
              >
                {t.toUpperCase()} Outline
              </Button>
            );
          })}
        </div>
        <Button block loading={loading} type="primary">
          Primary Block Button
        </Button>
        <div>--</div>
        <Button block loading={loading} type="info" outline>
          Info Block Outline Button
        </Button>
      </div>
    );
  }}
</Playground>
